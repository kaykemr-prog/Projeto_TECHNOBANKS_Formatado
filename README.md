CREATE DATABASE SEMAFOROS_INTELIGENTES;
USE SEMAFOROS_INTELIGENTES;

CREATE TABLE CRUZAMENTOS(
	ID_Cruzamento INT PRIMARY KEY AUTO_INCREMENT,
    Identificacao VARCHAR(100) NOT NULL,
    Regiao VARCHAR(100) NOT NULL
);

CREATE TABLE CONFIGURACAO_TEMPO(
	ID_Configuracao INT PRIMARY KEY AUTO_INCREMENT,
    Tempo_padrao INT NOT NULL,
    Tempo_min INT NOT NULL,
    Tempo_max INT NOT NULL
);

CREATE TABLE SEMAFOROS(
	ID_Semaforo INT PRIMARY KEY AUTO_INCREMENT,
    Localizacao VARCHAR(100) NOT NULL,
    Tipo ENUM('VEICULAR','PEDESTRE') NOT NULL,
    Status_atual ENUM('VERDE','AMARELO','VERMELHO') NOT NULL,
    Tempo_atual INT NOT NULL,
    ID_Cruzamento INT NOT NULL,
    ID_Configuracao INT NOT NULL,
    FOREIGN KEY (ID_Cruzamento) REFERENCES CRUZAMENTOS(ID_Cruzamento),
    FOREIGN KEY (ID_Configuracao) REFERENCES CONFIGURACAO_TEMPO(ID_Configuracao)
);

CREATE TABLE SENSORES(
	ID_Sensor INT PRIMARY KEY AUTO_INCREMENT,
	Tipo VARCHAR(100) NOT NULL,
	Status ENUM('ATIVO','INATIVO','MANUTENCAO','FALHA') NOT NULL,
    ID_Cruzamento INT NOT NULL,
	FOREIGN KEY (ID_Cruzamento) REFERENCES CRUZAMENTOS(ID_Cruzamento)
);

CREATE TABLE LEITURAS(
	ID_Leitura INT PRIMARY KEY AUTO_INCREMENT,
    Quantidade_veiculos INT NOT NULL,
    Velocidade_media INT NOT NULL,
    Data_hora DATETIME NOT NULL,
    ID_Sensor INT NOT NULL,
    FOREIGN KEY (ID_Sensor) REFERENCES SENSORES(ID_Sensor)
);

CREATE TABLE EVENTOS(
	ID_Evento INT PRIMARY KEY AUTO_INCREMENT,
    Tipo ENUM('CONGESTIONAMENTO','ACIDENTE','TRAFEGO_INTENSO', 'EMERGENCIA','FALHA_SENSOR') NOT NULL,
    Descricao VARCHAR(250),
    Prioridade ENUM('ALTO','MEDIO','BAIXO') NOT NULL,
    Data_hora DATETIME NOT NULL,
    ID_Leitura INT NOT NULL,
    ID_Cruzamento INT NOT NULL,
    FOREIGN KEY (ID_Leitura) REFERENCES LEITURAS(ID_Leitura),
    FOREIGN KEY (ID_Cruzamento) REFERENCES CRUZAMENTOS(ID_Cruzamento)
);
